Manual entry enabled."); } }); }); // Helper function to wipe the form clean function clearCaptureForm() { // Reset dropdown to first option (usually your first species) const dropdown = document.getElementById('breed-dropdown'); if (dropdown.options.length > 0) dropdown.selectedIndex = 0; document.getElementById('unit-input').value = ""; document.getElementById('born-date-input').value = ""; document.getElementById('last-wc-input').value = ""; document.getElementById('last-fed-input').value = ""; }
