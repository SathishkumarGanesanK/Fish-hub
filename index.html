// Scanner Logic - UPDATED for better auto-population const html5QrCode = new Html5Qrcode("reader"); html5QrCode.start({ facingMode: "environment" }, { fps: 15, qrbox: 250 }, (text) => { // 1. Set the Tank ID display document.getElementById('display-barcode').innerText = text; // 2. Open the form document.getElementById('capture-form').classList.remove('hidden'); // 3. Look up this Tank ID in our database db.scans.get(text).then(existing => { if(existing) { // AUTO-POPULATE: Match species name to the dropdown const breedDropdown = document.getElementById('breed-dropdown'); // We set the value and also trigger a change event to be safe breedDropdown.value = existing.breed || ""; document.getElementById('unit-input').value = existing.unit || ""; document.getElementById('born-date-input').value = existing.bornDate || ""; document.getElementById('last-wc-input').value = existing.lastWC || ""; document.getElementById('last-fed-input').value = existing.lastFed || ""; console.log("Found existing tank #" + text + ". Populated Species: " + existing.breed); } else { // Reset fields for a brand new tank scan document.getElementById('unit-input').value = ""; document.getElementById('born-date-input').value = ""; document.getElementById('last-wc-input').value = ""; document.getElementById('last-fed-input').value = ""; console.log("New Tank ID detected: " + text); } }).catch(err => { console.error("Database lookup error:", err); }); });
